//Base de datos MongoDB:
1. Actualizar paquetes:

sudo apt update && sudo apt upgrade -y

2. Importar clave publida de MongoDB:

curl -fsSL https://pgp.mongodb.com/server-7.0.asc | \
   sudo gpg -o /usr/share/keyrings/mongodb-server-7.0.gpg \
   --dearmor

3. Agregar el repositorio de MongoDB:

echo "deb [ arch=amd64,arm64 signed-by=/usr/share/keyrings/mongodb-server-7.0.gpg ] \
https://repo.mongodb.org/apt/ubuntu jammy/mongodb-org/7.0 multiverse" | \
sudo tee /etc/apt/sources.list.d/mongodb-org-7.0.list

4. Actualizar repositorio e instalar MongoDB:

sudo apt update
sudo apt install -y mongodb-org

5. Habilitar Conexiones publicas:

sudo nano /etc/mongod.conf

6. Cambiar Bindip de la seccion net:

net:
  port: 27017
  bindIp: 127.0.0.1

7. Remplazarlo con estos datos:

net:
  port: 27017
  bindIp: 0.0.0.0

8. Habilitar servicio de MongoDB:

sudo systemctl start mongod
sudo systemctl enable mongod
sudo systemctl status mongod

9. Probar la conexion desde PC local:

mongosh "mongodb://x.x.x.x:27017"
